17  package org.leplus.ristretto.util;
18  
19  import static org.junit.Assert.assertEquals;
20  import static org.junit.Assert.assertNotEquals;
21  import static org.junit.Assert.assertSame;
22  
23  import com.google.common.testing.EqualsTester;
24  import com.google.common.testing.SerializableTester;
25  import org.apache.commons.lang3.SerializationUtils;
26  import org.junit.Test;
27  
28  /**
29   * Tests for the {@link IdentityObject} class.
30   *
31   * @author Thomas Leplus
32   * @since 1.0.0
33   */
34  public final class TestIdentityObject {
35  
36    /** Checks that IdentityObject.IT is equal (==) to a shallow clone of itself. */
37    @Test
38    public void testEqualClone() {
39      assertSame(IdentityObject.IT, IdentityObject.IT.clone());
40    }
41  
42    /** Checks that IdentityObject.IT is equal (==) to a deep clone of itself. */
43    @Test
44    public void testEqualDeepClone() {
45      assertSame(IdentityObject.IT, SerializationUtils.clone(IdentityObject.IT));
46    }
47  
48    /** Checks that IdentityObject.IT equals() itself. */
49    @Test
50    public void testEquals() {
51      assertEquals(IdentityObject.IT, IdentityObject.IT);
52      assertNotEquals(IdentityObject.IT, new DuplicityObject());
53    }
54  
55    /** Checks that IdentityObject.IT passes guava's equality tests. */
56    @Test
57    public void testEqualsGuava() {
58      new EqualsTester().addEqualityGroup(IdentityObject.IT, IdentityObject.IT).testEquals();
59    }
60  
61    /** Checks that IdentityObject.IT equals() a shallow clone of itself. */
62    @Test
63    public void testEqualsClone() {
64      assertEquals(IdentityObject.IT, IdentityObject.IT.clone());
65      assertEquals(IdentityObject.IT.clone(), IdentityObject.IT);
66    }
67  
68    /** Checks that IdentityObject.IT equals() a deep clone of itself. */
69    @Test
70    public void testEqualsDeepClone() {
71      assertEquals(IdentityObject.IT, SerializationUtils.clone(IdentityObject.IT));
72      assertEquals(SerializationUtils.clone(IdentityObject.IT), IdentityObject.IT);
73    }
74  
75    /** Checks IdentityObject.IT's hashcode remains constant. */
76    @Test
77    public void testHashCode() {
78      assertEquals(IdentityObject.IT.hashCode(), IdentityObject.IT.hashCode());
79      assertEquals(IdentityObject.IT.hashCode(), IdentityObject.IT.clone().hashCode());
80      assertEquals(
81          IdentityObject.IT.hashCode(), SerializationUtils.clone(IdentityObject.IT).hashCode());
82    }
83  
84    /** Checks that IdentityObject.IT passes guava's serialization tests. */
85    @Test
86    public void testSerializeGuava() {
87      SerializableTester.reserializeAndAssert(IdentityObject.IT);
88    }
89  }
