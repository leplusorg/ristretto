41  /**
42   * Tests the {@link org.leplus.ristretto.util.IdentityHashSet} class.
43   *
44   * @author Thomas Leplus
45   * @since 1.0.0
46   */
47  @RunWith(Suite.class) @Suite.SuiteClasses({
48      TestIdentityHashSet.GuavaTests.class,
49      TestIdentityHashSet.AdditionalTests.class
50    })
51  public class TestIdentityHashSet {
52  
53    /**
54     * Some additional tests.
55     *
56     * @author Thomas Leplus
57     * @since 1.0.0
58     */
59    public static class AdditionalTests {
60  
61      /**
62       * Tests {@link IdentityHashSet#clear()}.
63       */
64      @Test
65      public void testClear() {
66        final Set<DuplicityObject> set = new IdentityHashSet<>();
67        final DuplicityObject d = new DuplicityObject();
68        set.add(d);
69        assertFalse(set.isEmpty());
70        assertEquals(1, set.size());
71        set.clear();
72        assertTrue(set.isEmpty());
73        assertEquals(0, set.size());
74      }
75  
76      /**
77       * Tests {@link IdentityHashSet#clone()}.
78       */
79      @Test
80      public void testClone() {
81        final IdentityHashSet<DuplicityObject> set1 = new IdentityHashSet<>();
82        set1.add(new DuplicityObject());
83        @SuppressWarnings("unchecked")
84        final IdentityHashSet<DuplicityObject> set2 =
85          (IdentityHashSet<DuplicityObject>) set1.clone();
86        assertEquals(set1, set2);
87        assertSame(set1.iterator().next(), set2.iterator().next());
88        set1.clear();
89        assertEquals(0, set1.size());
90        assertEquals(1, set2.size());
91      }
92  
93      /**
94       * Tests {@link IdentityHashSet#equals(Object)} with same object in two set.
95       */
96      @Test
97      public void testEqualsOneObject() {
98        final DuplicityObject d = new DuplicityObject();
99        final Set<DuplicityObject> set1 = new IdentityHashSet<>();
100       set1.add(d);
101       final Set<DuplicityObject> set2 = new IdentityHashSet<>();
102       set2.add(d);
103       assertEquals(set1, set2);
104     }
105 
106     /**
107      * Tests {@link IdentityHashSet#equals(Object)} with different object in two set.
108      */
109     @Test
110     public void testEqualsTwoObjects() {
111       final Set<DuplicityObject> set1 = new IdentityHashSet<>();
112       set1.add(new DuplicityObject());
113       final Set<DuplicityObject> set2 = new IdentityHashSet<>();
114       set2.add(new DuplicityObject());
115       assertNotEquals(set1, set2);
116     }
117 
118     /**
119      * Tests {@link IdentityHashSet#hashCode()}.
120      */
121     @Test
122     public void testHashCode() {
123       final DuplicityObject a = new DuplicityObject();
124       final DuplicityObject b = new DuplicityObject();
125       final DuplicityObject c = new DuplicityObject();
126       final Set<DuplicityObject> set1 =
127           new IdentityHashSet<>(Arrays.asList(a, b, c));
128       final Set<DuplicityObject> set2 =
129           new IdentityHashSet<>(Arrays.asList(b, c, a));
130       assertEquals(set1.hashCode(), set2.hashCode());
131     }
132 
133     /**
134      * Tests {@link IdentityHashSet#isEmpty()}.
135      */
136     @Test
137     public void testIsEmpty() {
138       final Set<DuplicityObject> set = new IdentityHashSet<>();
139       assertTrue(set.isEmpty());
140       assertEquals(0, set.size());
141     }
142 
143     /**
144      * Tests {@link IdentityHashSet#size()}.
145      */
146     @Test
147     public void testNotEmpty() {
148       final Set<DuplicityObject> set = new IdentityHashSet<>();
149       set.add(new DuplicityObject());
150       assertFalse(set.isEmpty());
151       assertEquals(1, set.size());
152     }
153 
154     /**
155      * Various tests with single object.
156      */
157     @Test
158     public void testOneObject() {
159       final Set<DuplicityObject> set = new IdentityHashSet<>();
160       final DuplicityObject d = new DuplicityObject();
161       set.add(d);
162       set.add(d);
163       assertFalse(set.isEmpty());
164       assertEquals(1, set.size());
165       assertTrue(set.contains(d));
166       assertFalse(set.contains(new DuplicityObject()));
167       set.remove(d);
168       assertFalse(set.contains(d));
169       assertTrue(set.isEmpty());
170       assertEquals(0, set.size());
171     }
172 
173     /**
174      * Various tests with two different objects.
175      */
176     @Test
177     public void testTwoObjects() {
178       final Set<DuplicityObject> set = new IdentityHashSet<>();
179       set.add(new DuplicityObject());
180       set.add(new DuplicityObject());
181       assertFalse(set.isEmpty());
182       assertEquals(2, set.size());
183       assertFalse(set.contains(new DuplicityObject()));
184     }
185 
186     /**
187      * Various equality tests.
188      */
189     @Test
190     public void testEquals() {
191       final DuplicityObject d = new DuplicityObject();
192       new EqualsTester()
193         .addEqualityGroup(new IdentityHashSet<DuplicityObject>(0),
194           new IdentityHashSet<DuplicityObject>()).testEquals();
195       new EqualsTester()
196         .addEqualityGroup(new IdentityHashSet<>(Arrays.asList(d)),
197           new IdentityHashSet<>(Arrays.asList(d))).testEquals();
198       new EqualsTester()
199         .addEqualityGroup(new IdentityHashSet<>(Arrays.asList(d, d, d)),
200           new IdentityHashSet<>(Arrays.asList(d, d, d))).testEquals();
201     }

205   /**
206    * Guava collection testing.
207    *
208    * @author Thomas Leplus
209    * @since 1.0.0
210    */
211   public static class GuavaTests {
212 
213     /**
214      * Creates the test suite.
215      * 
216      * @return the test suite
217      */
218     public static TestSuite suite() {
219       return SetTestSuiteBuilder.using(new TestStringSetGenerator() {
220 
221         /*
222          * (non-Javadoc)
223          *
224          * @see com.google.common.collect.testing.TestStringSetGenerator#create(java.lang. String[])
225          */
226         @Override
227         protected Set<String> create(final String[] elements) {
228           return new IdentityHashSet<>(Arrays.asList(elements));
229         }
230 
231       }).named("IdentityHashSet tests")
232           .withFeatures(SetFeature.GENERAL_PURPOSE,
233               CollectionFeature.ALLOWS_NULL_QUERIES,
234               CollectionFeature.ALLOWS_NULL_VALUES,
235               CollectionFeature.DESCENDING_VIEW,
236               CollectionFeature.FAILS_FAST_ON_CONCURRENT_MODIFICATION,
237               CollectionFeature.GENERAL_PURPOSE,
238               // Sets are not ordered
239               // CollectionFeature.KNOWN_ORDER,
240               // Duplicates used in this test are not ==
241               // CollectionFeature.REJECTS_DUPLICATES_AT_CREATION,
242               CollectionFeature.REMOVE_OPERATIONS,
243               CollectionFeature.RESTRICTS_ELEMENTS,
244               // Sets are not == after serialization
245               // CollectionFeature.SERIALIZABLE,
246               // Sets are not == after serialization
247               // CollectionFeature.SERIALIZABLE_INCLUDING_VIEWS,
248               CollectionFeature.SUBSET_VIEW,
249               CollectionFeature.SUPPORTS_ADD,
250               CollectionFeature.SUPPORTS_ITERATOR_REMOVE,
251               CollectionFeature.SUPPORTS_REMOVE,
252               CollectionSize.ANY,
253               CollectionSize.ONE,
254               CollectionSize.SEVERAL,
255               CollectionSize.ZERO)
256           .createTestSuite();
257     }
