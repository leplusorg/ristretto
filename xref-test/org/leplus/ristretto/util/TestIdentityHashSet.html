38  /**
39   * Tests the {@link org.leplus.ristretto.util.IdentityHashSet} class.
40   *
41   * @author Thomas Leplus
42   * @since 1.0.0
43   */
44  @RunWith(Suite.class)
45  @Suite.SuiteClasses({ TestIdentityHashSet.GuavaTests.class,
46      TestIdentityHashSet.AdditionalTests.class })
47  public class TestIdentityHashSet {
48  
49    /**
50     * Some additional tests.
51     *
52     * @author Thomas Leplus
53     * @since 1.0.0
54     */
55    public static class AdditionalTests {
56  
57      /**
58       * Tests {@link IdentityHashSet#clear()}.
59       */
60      @Test
61      public void testClear() {
62        final Set<DuplicityObject> set = new IdentityHashSet<>();
63        final DuplicityObject d = new DuplicityObject();
64        set.add(d);
65        assertFalse(set.isEmpty());
66        assertEquals(1, set.size());
67        set.clear();
68        assertTrue(set.isEmpty());
69        assertEquals(0, set.size());
70      }
71  
72      /**
73       * Tests {@link IdentityHashSet#clone()}.
74       */
75      @Test
76      public void testClone() {
77        final IdentityHashSet<DuplicityObject> set1 = new IdentityHashSet<>();
78        set1.add(new DuplicityObject());
79        @SuppressWarnings("unchecked")
80        final IdentityHashSet<DuplicityObject> set2 = (IdentityHashSet<DuplicityObject>) set1.clone();
81        assertEquals(set1, set2);
82        assertSame(set1.iterator().next(), set2.iterator().next());
83        set1.clear();
84        assertEquals(0, set1.size());
85        assertEquals(1, set2.size());
86      }
87  
88      /**
89       * Tests {@link IdentityHashSet#equals()} with same object in two set.
90       */
91      @Test
92      public void testEqualsOneObject() {
93        final DuplicityObject d = new DuplicityObject();
94        final Set<DuplicityObject> set1 = new IdentityHashSet<>();
95        set1.add(d);
96        final Set<DuplicityObject> set2 = new IdentityHashSet<>();
97        set2.add(d);
98        assertEquals(set1, set2);
99      }
100 
101     /**
102      * Tests {@link IdentityHashSet#equals()} with different object in two
103      * set.
104      */
105     @Test
106     public void testEqualsTwoObjects() {
107       final Set<DuplicityObject> set1 = new IdentityHashSet<>();
108       set1.add(new DuplicityObject());
109       final Set<DuplicityObject> set2 = new IdentityHashSet<>();
110       set2.add(new DuplicityObject());
111       assertNotEquals(set1, set2);
112     }
113 
114     /**
115      * Tests {@link IdentityHashSet#hashCode()}.
116      */
117     @Test
118     public void testHashCode() {
119       final DuplicityObject a = new DuplicityObject();
120       final DuplicityObject b = new DuplicityObject();
121       final DuplicityObject c = new DuplicityObject();
122       final Set<DuplicityObject> set1 = new IdentityHashSet<>(Arrays.asList(a, b, c));
123       final Set<DuplicityObject> set2 = new IdentityHashSet<>(Arrays.asList(b, c, a));
124       assertEquals(set1.hashCode(), set2.hashCode());
125     }
126 
127     /**
128      * Tests {@link IdentityHashSet#isEmpty()}.
129      */
130     @Test
131     public void testIsEmpty() {
132       final Set<DuplicityObject> set = new IdentityHashSet<>();
133       assertTrue(set.isEmpty());
134       assertEquals(0, set.size());
135     }
136 
137     /**
138      * Tests {@link IdentityHashSet#size()}.
139      */
140     @Test
141     public void testNotEmpty() {
142       final Set<DuplicityObject> set = new IdentityHashSet<>();
143       set.add(new DuplicityObject());
144       assertFalse(set.isEmpty());
145       assertEquals(1, set.size());
146     }
147 
148     /**
149      * Various tests with single object.
150      */
151     @Test
152     public void testOneObject() {
153       final Set<DuplicityObject> set = new IdentityHashSet<>();
154       final DuplicityObject d = new DuplicityObject();
155       set.add(d);
156       set.add(d);
157       assertFalse(set.isEmpty());
158       assertEquals(1, set.size());
159       assertTrue(set.contains(d));
160       assertFalse(set.contains(new DuplicityObject()));
161       set.remove(d);
162       assertFalse(set.contains(d));
163       assertTrue(set.isEmpty());
164       assertEquals(0, set.size());
165     }
166 
167     /**
168      * Various tests with two different objects.
169      */
170     @Test
171     public void testTwoObjects() {
172       final Set<DuplicityObject> set = new IdentityHashSet<>();
173       set.add(new DuplicityObject());
174       set.add(new DuplicityObject());
175       assertFalse(set.isEmpty());
176       assertEquals(2, set.size());
177       assertFalse(set.contains(new DuplicityObject()));
178     }
179 
180     /**
181      * Various equality tests.
182      */
183     @Test
184     public void testEquals() {
185       final DuplicityObject d = new DuplicityObject();
186       new EqualsTester().addEqualityGroup(new IdentityHashSet<DuplicityObject>(0),
187           new IdentityHashSet<DuplicityObject>()).testEquals();
188       new EqualsTester().addEqualityGroup(new IdentityHashSet<>(Arrays.asList(d)),
189           new IdentityHashSet<>(Arrays.asList(d))).testEquals();
190       new EqualsTester().addEqualityGroup(new IdentityHashSet<>(Arrays.asList(d, d, d)),
191           new IdentityHashSet<>(Arrays.asList(d, d, d))).testEquals();
192     }

196   /**
197    * Guava collection testing.
198    *
199    * @author Thomas Leplus
200    * @since 1.0.0
201    */
202   public static class GuavaTests {
203 
204     /**
205      * Creates the test suite.
206      */
207     public static TestSuite suite() {
208       return SetTestSuiteBuilder.using(new TestStringSetGenerator() {
209 
210         /*
211          * (non-Javadoc)
212          * 
213          * @see com.google.common.collect.testing.TestStringSetGenerator#create(java.lang. String[])
214          */
215         @Override
216         protected Set<String> create(final String[] elements) {
217           return new IdentityHashSet<>(Arrays.asList(elements));
218         }
219 
220       }).named("IdentityHashSet tests")
221           .withFeatures(SetFeature.GENERAL_PURPOSE, CollectionFeature.ALLOWS_NULL_QUERIES,
222               CollectionFeature.ALLOWS_NULL_VALUES, CollectionFeature.DESCENDING_VIEW,
223               CollectionFeature.FAILS_FAST_ON_CONCURRENT_MODIFICATION,
224               CollectionFeature.GENERAL_PURPOSE,
225               // CollectionFeature.KNOWN_ORDER, // Sets are not ordered
226               // CollectionFeature.REJECTS_DUPLICATES_AT_CREATION, // Duplicates used in this test
227               // are not ==
228               CollectionFeature.REMOVE_OPERATIONS, CollectionFeature.RESTRICTS_ELEMENTS,
229               // CollectionFeature.SERIALIZABLE, // Sets are not == after serialization
230               // CollectionFeature.SERIALIZABLE_INCLUDING_VIEWS, // Sets are not == after
231               // serialization
232               CollectionFeature.SUBSET_VIEW, CollectionFeature.SUPPORTS_ADD,
233               CollectionFeature.SUPPORTS_ITERATOR_REMOVE, CollectionFeature.SUPPORTS_REMOVE,
234               CollectionSize.ANY, CollectionSize.ONE, CollectionSize.SEVERAL, CollectionSize.ZERO)
235           .createTestSuite();
236     }
